# Add this to your projects composer.json:

```json
  "extra": {
    "grumphp": {
      "config-default-path": "vendor/pluswerk/grumphp-config/grumphp.yml"
    }
  },
  "require-dev": {
    "pluswerk/grumphp-config": "2.*"
  }
```
